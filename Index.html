    <script>

        let currentLang = 'es'; // Idioma por defecto


        function changeLang(lang) {

            currentLang = lang;

            

            // 1. Cambiar todos los textos con data-attributes

            const elements = document.querySelectorAll('[data-es]');

            elements.forEach(el => {

                el.innerHTML = el.getAttribute(`data-${lang}`);

            });


            // 2. Intercambiar bloques de servicios (ES/EN)

            if (lang === 'es') {

                document.querySelectorAll('.lang-es').forEach(el => el.classList.remove('lang-hidden'));

                document.querySelectorAll('.lang-en').forEach(el => el.classList.add('lang-hidden'));

                document.getElementById('btn-es').className = "px-4 py-1.5 rounded-md bg-white shadow-sm text-blue-600 transition uppercase font-bold";

                document.getElementById('btn-en').className = "px-4 py-1.5 rounded-md text-slate-500 transition uppercase font-bold";

            } else {

                document.querySelectorAll('.lang-es').forEach(el => el.classList.add('lang-hidden'));

                document.querySelectorAll('.lang-en').forEach(el => el.classList.remove('lang-hidden'));

                document.getElementById('btn-en').className = "px-4 py-1.5 rounded-md bg-white shadow-sm text-blue-600 transition uppercase font-bold";

                document.getElementById('btn-es').className = "px-4 py-1.5 rounded-md text-slate-500 transition uppercase font-bold";

            }

        }


        function contactWA(tipo) {

            let msg = "";

            const phone = "523223537716";


            if (currentLang === 'es') {

                // Mensajes en ESPAÑOL

                if (tipo === 'despacho') msg = "Hola Lic. Guzmán, requiero apoyo de corresponsalía para mi despacho.";

                else if (tipo === 'particular') msg = "Hola Lic. Guzmán, necesito ayuda con un trámite administrativo.";

                else msg = "Hola, me interesa una asesoría. Mis datos son: \n\nNombre: \nServicio de interés:";

            } else {

                // Mensajes en INGLÉS

                if (tipo === 'despacho') msg = "Hello Mr. Guzman, I am interested in legal correspondent support for my firm.";

                else if (tipo === 'particular') msg = "Hello Mr. Guzman, I need assistance with administrative paperwork.";

                else msg = "Hello, I am interested in legal support. My information: \n\nName: \nService of interest:";

            }


            // Abrir WhatsApp con el mensaje codificado según el idioma activo

            window.open(`https://wa.me/${phone}?text=${encodeURIComponent(msg)}`, '_blank');

        }
